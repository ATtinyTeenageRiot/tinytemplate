54 void ticksInit() {
55  // Stop timer and clear prescaler
56  TCCR1 = 0;
57  TCNT1 = 0;
58  GTCCR &= 0x81;
59  GTCCR |= (1 << PSR1);
60  // Set up the prescaler and enable overflow interrupt

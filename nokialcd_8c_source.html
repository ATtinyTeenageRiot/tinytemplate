53 void lcdInit() {
54  // Set up the output pins, ensuring they are all 'low' to start
55  uint8_t val = (1 << LCD_SCK) | (1 << LCD_MOSI) | (1 << LCD_RESET) | (1 << LCD_CD);
56  PORTB &= ~val;
57  DDRB |= val;
58  // Do a hard reset on the LCD
59  wait(10);
60  PORTB |= (1 << LCD_RESET);
61  // Initialise the LCD
62  lcdCommand(0x21); // LCD Extended Commands.
63  lcdCommand(0xA1); // Set LCD Vop (Contrast).
64  lcdCommand(0x04); // Set Temp coefficent.
65  lcdCommand(0x14); // LCD bias mode 1:48.
66  lcdCommand(0x20); // LCD Normal commands
67  lcdCommand(0x0C); // Normal display, horizontal addressing

121 void lcdPrintChar(uint8_t row, uint8_t col, char ch, bool invert) {
122  // Make sure it is on the screen
123  if((row>=LCD_ROW)||(col>=LCD_COL))
124  return;
125  // If the character is invalid replace it with the '?'
126  if((ch<0x20)||(ch>0x7f))
127  ch = '?';
128  // Set the starting address
129  lcdCommand(0x80 | col);
130  lcdCommand(0x40 | (row % LCD_ROW));
131  // And send the column data
132  const uint8_t *chdata = BASE_FONT + ((ch - 0x20) * 5);
133  for(uint8_t pixels = 0; (pixels < CHAR_WIDTH) && (col < LCD_COL); pixels++, col++, chdata++) {
134  uint8_t data = pgm_read_byte_near(chdata);
135  lcdData(invert?~data:data);
136  }
137  // Add the padding byte
138  if(col < LCD_COL)
139  lcdData(invert?0xFF:0x00);
140  }

212 void lcdImageP(uint8_t row, uint8_t col, const uint8_t *img, bool invert) {
213  // Break out the width and height
214  uint8_t data = pgm_read_byte_near(img++);
215  uint8_t height = (data >> 6) + 1;
216  uint8_t width = (data & 0x03F) + 1;
217  while((height>0)&&(row<LCD_ROW)) {
218  // Set the starting address
219  lcdCommand(0x80 | col);
220  lcdCommand(0x40 | row);
221  // Send the data (up to the end of the screen)
222  for(uint8_t offset=0; offset<width; offset++, img++) {
223  if((col+offset)<LCD_COL) {
224  if(invert)
225  lcdData(~pgm_read_byte_near(img));
226  else
227  lcdData(pgm_read_byte_near(img));
228  }
229  }
230  height--;
231  row++;
232  }
233  }

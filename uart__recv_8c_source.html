74  // Move everything down
75  for(uint8_t index=0; index<g_index; g_buffer[index] = g_buffer[index + 1], index++);
76  // Done
77  sei();

80  DDRB &= ~(1 << UART_RX);

82  // Read the byte
83  cli();

85  " ldi r18, %[rxdelay2] \n\t" // 1.5 bit delay
86  " ldi %0, 0x80 \n\t" // bit shift counter

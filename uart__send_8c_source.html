39 void uartSend(char ch) {
40  // Set to output state and bring high
41  UART_PORT |= (1 << UART_PIN);
42  UART_DDR |= (1 << UART_PIN);
43  asm volatile(
44  " cbi %[uart_port], %[uart_pin] \n\t" // start bit
45  " in r0, %[uart_port] \n\t"
46  " ldi r30, 3 \n\t" // stop bit + idle state
47  " ldi r28, %[txdelay] \n\t"
48  "TxLoop: \n\t"
49  // 8 cycle loop + delay - total = 7 + 3*r22
50  " mov r29, r28 \n\t"
51  "TxDelay: \n\t"
52  // delay (3 cycle * delayCount) - 1
53  " dec r29 \n\t"
54  " brne TxDelay \n\t"
55  " bst %[ch], 0 \n\t"
56  " bld r0, %[uart_pin] \n\t"
57  " lsr r30 \n\t"
58  " ror %[ch] \n\t"
59  " out %[uart_port], r0 \n\t"
60  " brne TxLoop \n\t"
61  :
62  : [uart_port] "I" (_SFR_IO_ADDR(UART_PORT)),
63  [uart_pin] "I" (UART_PIN),
64  [txdelay] "I" (TXDELAY),
65  [ch] "r" (ch)
66  : "r0","r28","r29","r30");
67  // Change back to idle state
68  UART_DDR &= ~(1 << UART_PIN);
69  UART_PORT &= ~(1 << UART_PIN);
70  }
